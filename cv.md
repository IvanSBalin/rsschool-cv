``` 
function solve(s){
    let sum = 0;
     for (let i = 0; i < s.length; i++) {
        if (s[i] == s[i].toUpperCase()) {
        sum += 1;
        } else {
        sum -= 1;
        }
    }
return sum > 0 ? s.toUpperCase() : s.toLowerCase(); 
}
```
